Code Examples

Gradient Backgrounds

<!-- Simple two-color gradient -->
<div class="bg-gradient-to-r from-cyan-500 to-blue-500">
  Content here
</div>

<!-- Multi-color gradient -->
<div class="bg-gradient-to-r from-green-400 via-blue-500 to-purple-600">
  Content here
</div>

<-- Direction modifiers -->
to-r (right), to-l (left), to-t (top),
to-b (bottom), to-tr (top right), etc.

Image Backgrounds

<!-- Using arbitrary value -->
<div class="bg-[url('/path/to/image.jpg')]">
  Content here
</div>

<!-- Background size -->
bg-cover, bg-contain, bg-auto

<!-- Background position -->
bg-center, bg-top, bg-left, etc.

<!-- Background repeat -->
bg-repeat, bg-no-repeat, bg-repeat-x
